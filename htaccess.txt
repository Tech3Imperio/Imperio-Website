RewriteBase /

# If the request is not for a file
RewriteCond %{REQUEST_FILENAME} !-f
# If the request is not for a directory
RewriteCond %{REQUEST_FILENAME} !-d
# Redirect everything else to index.html
RewriteRule ^(.*)$ /index.html [L]
